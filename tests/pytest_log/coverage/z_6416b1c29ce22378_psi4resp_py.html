49 xyz: str, 

50 method: str = "wB97X", 

51 basis: str = "def2-m_svp", 

52 method_esp: str = "wB97X", 

53 basis_esp: str = "def2-SVPD",

55 optimize: bool = True, 

56 runresp: bool = True, 

57 psi4outfile: str = "psi4.out", 

58 **kwargs: Any 

59) -> None: 

60 """ 

61 Perform single point calculation and ESP fitting using Psi4 and RESP. 

62 

63 Args: 

64 xyz (str): Path to the xyz file. 

65 method (str): The method for the single point calculation. 

66 basis (str): The basis set for the single point calculation. 

67 method_esp (str): The method for the ESP fitting. 

68 basis_esp (str): The basis set for the ESP fitting. 

69 usepcm (bool): Whether to use PCM for ESP fitting. 

70 optimize (bool): Whether to perform geometry optimization. 

71 runresp (bool): Whether to perform ESP fitting. 

72 psi4outfile (str): The output file for Psi4. 

73 **kwargs: Additional options. 

74 

75 Returns: 

76 None 

77 """ 

78 

79 p4mol = psi4.geometry(xyz) 

80 p4mol.update_geometry() 

81 

82 psi4.core.set_output_file(psi4outfile, False) 

83 if "threads" in kwargs: 

84 psi4.set_num_threads(int(kwargs["threads"])) 

85 else: 

86 ncpus = len(os.sched_getaffinity(0)) 

87 psi4.set_num_threads(ncpus) 

88 if "memory" in kwargs: 

89 psi4.set_memory(str(kwargs["memory"])) 

90 else: 

91 psi4.set_memory("1GiB")

93 if optimize: 

94 print(f"===> Geometry optimization by psi4 module using {method}/{basis}") 

95 psi4.set_options({"G_CONVERGENCE": "NWCHEM_LOOSE", "basis": basis}) 

96 psi4.optimize(method) 

97 __savegeom(p4mol) 

98 else: 

99 print("Info: Skip geometry optimization.") 

100 

101 if runresp is not True: 

102 return 

103 

104 print(f"===> Single point calc. for ESP grid by psi4 module using {method_esp}/{basis_esp}") 

105 if usepcm is True: 

106 p4mol.reset_point_group("C1") 

107 psi4.set_options({"pcm": "true", "pcm_scf_type": "total"}) 

108 dummy = "pcmsolver.inp" 

109 # if not os.path.isfile(pcmfile):

133def __savegeom(p4mol: psi4.core.Molecule, outfile: str = "optgeom.xyz") -> None: 

134 """Save the optimized geometry in the XYZ format. 

135 

136 Parameters 

137 ---------- 

138 p4mol : psi4.core.Molecule 

139 The molecule object in the Psi4 format. 

140 outfile : str, optional 

141 The name of the output file. Default is "optgeom.xyz". 

142 

143 Returns 

144 ------- 

145 None 

146 """ 

147 natom = p4mol.natom() 

148 geom, _, elem, _, _ = p4mol.to_arrays() 

149 charge = p4mol.molecular_charge() 

150 mult = p4mol.multiplicity() 

151 

152 geom *= __BOHR_TO_ANGSTROM 

153 with open(outfile, "w", encoding="utf-8") as f: 

154 f.write(f"{natom}\n") 

155 f.write(f"{charge} {mult}\n") 

156 temp = "" 

157 for e, g in zip(elem, geom): 

158 temp += f"{e:<8} {g[0]:>12.5f} {g[1]:>12.5f} {g[2]:>12.5f}\n" 

159 f.write(temp)

162def __convertgesp( 

163 p4mol: psi4.core.Molecule, 

164 esp: List[float], 

165 outfile: str = "antechamber.gesp", 

166 xyzfile: str = "1_default_grid.dat", 

167 efgfile: str = "1_default_grid_esp.dat", 

168) -> None: 

169 """ 

170 Convert the geometry and ESP charges to the ESP file format. 

171 

172 Parameters 

173 ---------- 

174 p4mol : psi4.core.Molecule 

175 The molecule object in the Psi4 format. 

176 esp : List[float] 

177 The ESP charges. 

178 outfile : str, optional 

179 The name of the output file. Default is "antechamber.gesp". 

180 xyzfile : str, optional 

181 The name of the file containing the grid coordinates. Default is "1_default_grid.dat". 

182 efgfile : str, optional 

183 The name of the file containing the ESP values. Default is "1_default_grid_esp.dat". 

184 

185 Returns 

186 ------- 

187 None 

188 """ 

189 natom = p4mol.natom() 

190 geom, _, elem, _, _ = p4mol.to_arrays() 

191 charge = p4mol.molecular_charge() 

192 mult = p4mol.multiplicity() 

193 

194 xyzfit = np.loadtxt(xyzfile) 

195 xyzfit /= __BOHR_TO_ANGSTROM 

196 efgfit = np.loadtxt(efgfile) 

197 nfit = len(xyzfit) 

198 

199 with open(outfile, "w", encoding="utf-8") as f: 

200 f.write(" ESP FILE - ATOMIC UNITS\n") 

201 f.write(f" CHARGE = {charge:3} - MULTIPLICITY = {mult:3}\n") 

202 f.write(f" ATOMIC COORDINATES AND ESP CHARGES. #ATOMS = {natom:8}\n") 

203 for i in range(natom): 

204 f.write( 

205 f" {elem[i]:<2s} {geom[i][0]:16.8E}{geom[i][1]:16.8E}" 

206 f"{geom[i][2]:16.8E}{esp[i]:16.8E}\n" 

207 ) 

208 f.write(" DIPOLE MOMENT:\n") 

209 f.write(f" X={0.0:16.8E} Y={0.0:16.8E} Z={0.0:16.8E} Total={0.0:16.8E}\n") 

210 f.write(" TRACELESS QUADRUPOLE MOMENT:\n") 

211 f.write( 

212 f" XX={0.0:16.8E} YY={0.0:16.8E} ZZ={0.0:16.8E}\n" 

213 ) 

214 f.write( 

215 f" XY={0.0:16.8E} XZ={0.0:16.8E} YZ={0.0:16.8E}\n" 

216 ) 

217 f.write(f" ESP VALUES AND GRID POINT COORDINATES. #POINTS ={nfit:8}\n") 

218 for i in range(nfit): 

219 f.write( 

220 f"{efgfit[i]:16.8E}{xyzfit[i][0]:16.8E}{xyzfit[i][1]:16.8E}{xyzfit[i][2]:16.8E}\n" 

221 )

224def __readfile(fn: str, charge: int, multiplicity: int) -> str: 

225 """ 

226 Reads a file and returns its contents. 

227 

228 Parameters 

229 ---------- 

230 fn : str 

231 The name of the file. 

232 charge : int 

233 The charge of the molecule. 

234 multiplicity : int 

235 The multiplicity of the molecule. 

236 

237 Returns 

238 ------- 

239 str 

240 The contents of the file. 

241 """ 

242 _, ext = os.path.splitext(fn) 

243 ext = ext[1:] 

244 

245 if ext == "xyz": 

246 gxyz = "" 

247 with open(fn, encoding="utf-8") as f: 

248 natom = int(f.readline()) 

249 line2 = f.readline() 

250 if line2 != "\n": 

251 _charge, _mult = line2.strip().split() 

252 assert isinstance(_charge, str) 

253 assert isinstance(_mult, str) 

254 gxyz = f"{int(_charge)} {int(_mult)}\n" 

255 

256 elif charge is not None and multiplicity is not None: 

257 gxyz = f"{charge} {multiplicity}\n" 

258 else: 

259 gxyz = "0 1\n" 

260 

261 for _ in range(natom): 

262 gxyz += f.readline() 

263 

264 else: 

265 raise TypeError("Error: non-supported file type") 

266 

267 return gxyz
